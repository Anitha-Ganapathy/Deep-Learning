# Generator network
generator = Sequential()


#hidden layer 1

generator.add(LeakyReLU(alpha=0.2))
generator.add(BatchNormalization())

# Hidden layer 2
generator.add(Dense(400,kernel_initializer=init))
generator.add(LeakyReLU(alpha=0.2))

# Output layer 
generator.add(Dense(img_dim, activation='tanh'))

num_classes = 10

# Create label one hot encoding
label = Input(shape=(1,), dtype='int32')

z = Input(shape=(latent_dim,))

# Concatenate inputs (z x label)
input_generator = concatenate([z, label_onehot])

# Output image
img = generator(input_generator)

# Generator with condition input
generator = Model([z, label], img)

generator.summary()

class TimeHistory(keras.callbacks.Callback):
    def on_train_begin(self, logs={}):
        self.times = []

    def on_epoch_begin(self, batch, logs={}):
        self.epoch_time_start = time.time()

    def on_epoch_end(self, batch, logs={}):
        self.times.append(time.time() - self.epoch_time_start)

batch_size = 100

epochs = 300

real = np.ones(shape=(batch_size, 1))
fake = np.zeros(shape=(batch_size, 1))

d_loss = []
d_g_loss = []
acc_real1 = []
acc_fake1 = []

for e in range(epochs + 1):
    for i in range(len(X_train) // batch_size):
        
        # Train Discriminator weights
        discriminator.trainable = True
        
        # Real samples
        X_batch = X_train[i*batch_size:(i+1)*batch_size]
        real_labels = y_train[i*batch_size:(i+1)*batch_size].reshape(-1, 1)
        
        d_loss_real,acc_real = discriminator.train_on_batch(x=[X_batch, real_labels], y=real)
        
        # Fake Samples
        z = np.random.normal(0, 1, size=(batch_size, latent_dim))
        random_labels = np.random.randint(0, 10, batch_size).reshape(-1, 1)
        X_fake = generator.predict_on_batch([z, random_labels])
        
        d_loss_fake,acc_fake = discriminator.train_on_batch(x=[X_fake, random_labels], y=fake)
         
        # Discriminator loss
        d_loss_batch = 0.5 * np.add(d_loss_real, d_loss_fake)
        
        # Train Generator weights
        discriminator.trainable = False
        
        z = np.random.normal(0, 1, size=(batch_size, latent_dim))
        random_labels = np.random.randint(0, 10, batch_size).reshape(-1, 1)
        d_g_loss_batch = d_g.train_on_batch(x=[z, random_labels], y=real)
   
        print(
            'epoch = %d/%d, batch = %d/%d,acc_real=%.2f,acc_fake=%.2f, d_loss=%.3f, g_loss=%.3f'  % (e + 1, epochs, i, len(X_train) // batch_size,acc_real,acc_fake, d_loss_batch, d_g_loss_batch[0]),
            100*' ',
            end='\r'
        )
    
    d_loss.append(d_loss_batch)
    d_g_loss.append(d_g_loss_batch[0])
    acc_real1.append(acc_real)
    acc_fake1.append(acc_fake)
    print('epoch = %d/%d,acc_real=%.2f,acc_fake=%.2f, d_loss=%.3f, g_loss=%.3f' % (e + 1, epochs,acc_real1[-1],acc_fake1[-1],d_loss[-1], d_g_loss[-1]), 100*' ')

    if e % 50 == 0:
        samples = 10
        z = np.random.normal(loc=0, scale=1, size=(samples, latent_dim))
        labels = np.arange(0, 10).reshape(-1, 1)
        
        x_fake = generator.predict([z, labels])

        for k in range(samples):
            plt.subplot(2, 5, k+1)
            plt.imshow(x_fake[k].reshape(28, 28), cmap='gray')
            plt.xticks([])
            plt.yticks([])

        plt.tight_layout()
        plt.show()
